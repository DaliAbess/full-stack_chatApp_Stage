###0.add .env file on backend 
MONGODB_URI=mongodb://mongoadmin:secret@localhost:27017/dbname?authSource=admin

###1run docker mongo db local
 docker run -d \
  -p 27017:27017 \
  --name mongo \
  -e MONGO_INITDB_ROOT_USERNAME=mongoadmin \
  -e MONGO_INITDB_ROOT_PASSWORD=secret \
  mongo:latest
